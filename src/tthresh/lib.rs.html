59pub fn compress<T: Elem>(
60    data: &[T],
61    shape: &[usize],
62    target: Target,
63    verbose: bool,
64    debug: bool,
65) -> Vec<u8> {
66    assert!(shape.len() >= 3);
67    assert_eq!(shape.iter().copied().product::<usize>(), data.len());
68    let shape = shape
69        .iter()
70        .copied()
71        .map(u32::try_from)
72        .collect::<Result<Vec<_>, _>>()
73        .unwrap();
74
75    let mut output = std::ptr::null_mut();
76    let mut noutput = 0;
77
78    #[allow(unsafe_code)]
79    unsafe {
80        tthresh_sys::my_compress(
81            shape.as_ptr(),
82            shape.len(),
83            data.as_ptr().cast::<c_char>(),
84            T::IO_TYPE,
85            std::ptr::from_mut(&mut output),
86            std::ptr::from_mut(&mut noutput),
87            match target {
88                Target::Eps(_) => tthresh_sys::Target_eps,
89                Target::RMSE(_) => tthresh_sys::Target_rmse,
90                Target::PSNR(_) => tthresh_sys::Target_psnr,
91            },
92            match target {
93                Target::Eps(v) | Target::RMSE(v) | Target::PSNR(v) => v,
94            },
95            verbose,
96            debug,
97        );
98    }
99
100    let mut compressed = vec![0_u8; noutput];
101
102    #[allow(unsafe_code)]
103    unsafe {
104        std::ptr::copy_nonoverlapping(output.cast::<u8>(), compressed.as_mut_ptr(), noutput);
105    }
106
107    #[allow(unsafe_code)]
108    unsafe {
109        tthresh_sys::dealloc_bytes(output);
110    }
111
112    compressed
113}

115pub fn decompress(
116    shape: &[u32],
117    compressed_file: &Path,
118    decompressed_file: &Path,
119    verbose: bool,
120    debug: bool,
121) {
122    let compressed_file = CString::new(compressed_file.as_os_str().as_encoded_bytes()).unwrap();
123    let decompressed_file = CString::new(decompressed_file.as_os_str().as_encoded_bytes()).unwrap();
124
125    #[allow(unsafe_code)]
126    unsafe {
127        tthresh_sys::my_decompress(
128            shape.as_ptr(),
129            shape.len(),
130            compressed_file.as_ptr(),
131            decompressed_file.as_ptr(),
132            verbose,
133            debug,
134        );
135    }
136}
