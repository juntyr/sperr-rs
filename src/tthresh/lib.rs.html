5use std::ffi::{CString, NulError};

7use tthresh_sys::my_main;

9pub fn tthresh(argv: &[&str]) {
10    let mut argv = argv
11        .iter()
12        .copied()
13        .map(|a| Ok(CString::new(a)?.into_raw()))
14        .collect::<Result<Vec<_>, NulError>>()
15        .unwrap();
16
17    #[allow(unsafe_code)]
18    let result = unsafe { my_main(argv.len().try_into().unwrap(), argv.as_mut_ptr()) };
19
20    assert_eq!(result, 0);
21}
